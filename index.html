Annotations

Handle Exceptions

public @interface HandleException {
    Class<? extends Throwable>[] exceptions() default { RuntimeException.class };
}

@HandleException
public void throwAndHandleException() {
    throw new IllegalArgumentException();
}

Suppress Exceptions

public @interface SuppressExceptions {
    Class<? extends Throwable>[] value() default { Throwable.class };
    Class<?> returnClass() default Null.class;
}

@SuppressExceptions
public void throwAndSuppressException() {
    throw new IllegalArgumentException();
}

Suppress throwed Exceptions

@SuppressExceptions(returnClass = ArrayList.class)
public List<?> throwAndSuppressExceptionReturnEmptyList() {
    throw new IllegalArgumentException();
}

Suppress Exception and return new instance of "returnClass"

Wrap Exceptions

public @interface WrapException {
    Class<? extends Throwable> catchException();
    Class<? extends Throwable> throwAs();
    String message() default "Wrapped exception";
    Class<? extends ExceptionWrapper<?, ?>> using() default DefaultExceptionWrapper.class;
}

@WrapException(catchException = IllegalArgumentException.class, throwAs = BusinessExcpetion.class)
public void throwAndWrapExeption() {
    throw new IllegalArgumentException();
}

Wraps Runtime-Exceptions to BussinessException
